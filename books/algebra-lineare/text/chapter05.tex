\chapter[Autovalori]{Autovalori e Diagonalizzazione}
\label{ch:autovalori}

Dato un endomorfismo $T \colon V \to V$ (o equivalentemente una matrice quadrata $A$), un problema centrale è trovare una base rispetto alla quale la matrice associata sia il più semplice possibile. La forma più semplice è quella diagonale, e gli autovalori sono la chiave per ottenerla.

\section{Autovalori e autovettori}
\label{sec:autovalori-def}

\begin{definitionbox}[Autovalore e autovettore]
Sia $A \in \mathcal{M}_{n \times n}(\mathbb{R})$. Uno scalare $\lambda \in \mathbb{R}$ è un \textbf{autovalore} di $A$ se esiste un vettore $\mathbf{v} \neq \mathbf{0}$ tale che:
\[ A\mathbf{v} = \lambda\mathbf{v}. \]
Il vettore $\mathbf{v}$ si chiama \textbf{autovettore} associato a $\lambda$. L'insieme di tutti gli autovettori associati a $\lambda$, insieme al vettore nullo, forma l'\textbf{autospazio} di $\lambda$:
\[ V_\lambda = \ker(A - \lambda I) = \{\mathbf{v} \in \mathbb{R}^n : A\mathbf{v} = \lambda\mathbf{v}\}. \]
\end{definitionbox}

\begin{notebox}[Interpretazione geometrica]
Un autovettore $\mathbf{v}$ di $A$ è un vettore la cui \emph{direzione} non viene modificata dalla trasformazione: $A$ agisce su $\mathbf{v}$ semplicemente ``allungandolo'' (o comprimendolo, o invertendolo) del fattore $\lambda$.
\end{notebox}

\section{Polinomio caratteristico}
\label{sec:polinomio-caratteristico}

\begin{theorembox}[Caratterizzazione degli autovalori]
$\lambda$ è autovalore di $A$ se e solo se $\det(A - \lambda I) = 0$.
\end{theorembox}
\begin{proof}
$\lambda$ è autovalore $\Leftrightarrow$ esiste $\mathbf{v} \neq \mathbf{0}$ con $(A - \lambda I)\mathbf{v} = \mathbf{0}$ $\Leftrightarrow$ $\ker(A - \lambda I) \neq \{\mathbf{0}\}$ $\Leftrightarrow$ $A - \lambda I$ non è invertibile $\Leftrightarrow$ $\det(A - \lambda I) = 0$.
\end{proof}

\begin{definitionbox}[Polinomio caratteristico]
Il \textbf{polinomio caratteristico} di $A \in \mathcal{M}_{n \times n}(\mathbb{R})$ è:
\[ p_A(\lambda) = \det(A - \lambda I). \]
È un polinomio di grado $n$ in $\lambda$. I suoi zeri reali sono gli autovalori di $A$.
\end{definitionbox}

\begin{examplebox}[Calcolo degli autovalori]
Sia $A = \begin{pmatrix} 4 & 1 \\ 2 & 3 \end{pmatrix}$. Il polinomio caratteristico è:
\begin{align*}
p_A(\lambda) &= \det\begin{pmatrix} 4 - \lambda & 1 \\ 2 & 3 - \lambda \end{pmatrix} = (4 - \lambda)(3 - \lambda) - 2 \\
&= \lambda^2 - 7\lambda + 10 = (\lambda - 2)(\lambda - 5).
\end{align*}
Gli autovalori sono $\lambda_1 = 2$ e $\lambda_2 = 5$.

\textbf{Autospazio di $\lambda_1 = 2$:} $(A - 2I)\mathbf{v} = \mathbf{0}$:
\[ \begin{pmatrix} 2 & 1 \\ 2 & 1 \end{pmatrix} \begin{pmatrix} v_1 \\ v_2 \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \end{pmatrix} \quad \Longrightarrow \quad v_1 = -\frac{v_2}{2}. \]
$V_2 = \text{Span}\big((-1, 2)\big)$.

\textbf{Autospazio di $\lambda_2 = 5$:} $(A - 5I)\mathbf{v} = \mathbf{0}$:
\[ \begin{pmatrix} -1 & 1 \\ 2 & -2 \end{pmatrix} \begin{pmatrix} v_1 \\ v_2 \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \end{pmatrix} \quad \Longrightarrow \quad v_1 = v_2. \]
$V_5 = \text{Span}\big((1, 1)\big)$.
\end{examplebox}

\begin{definitionbox}[Molteplicità]
Sia $\lambda_0$ autovalore di $A$:
\begin{itemize}
    \item La \textbf{molteplicità algebrica} $m_a(\lambda_0)$ è la molteplicità di $\lambda_0$ come radice di $p_A(\lambda)$.
    \item La \textbf{molteplicità geometrica} $m_g(\lambda_0) = \dim V_{\lambda_0} = \dim\ker(A - \lambda_0 I)$.
\end{itemize}
Vale sempre $1 \leq m_g(\lambda_0) \leq m_a(\lambda_0)$.
\end{definitionbox}

\section{Diagonalizzabilità}
\label{sec:diagonalizzabilita}

\begin{definitionbox}[Matrice diagonalizzabile]
$A \in \mathcal{M}_{n \times n}(\mathbb{R})$ è \textbf{diagonalizzabile} se esiste una matrice invertibile $P$ tale che:
\[ P^{-1}AP = D = \begin{pmatrix} \lambda_1 & & \\ & \ddots & \\ & & \lambda_n \end{pmatrix} \]
dove $\lambda_1, \ldots, \lambda_n$ sono gli autovalori di $A$ (con ripetizioni). Le colonne di $P$ sono gli autovettori corrispondenti.
\end{definitionbox}

\begin{theorembox}[Indipendenza degli autovettori]
\label{thm:autovettori-indipendenti}
Autovettori associati ad autovalori \emph{distinti} sono linearmente indipendenti.
\end{theorembox}

\section{Criteri di diagonalizzazione}
\label{sec:criteri-diag}

\begin{theorembox}[Condizione necessaria e sufficiente]
\label{thm:diag-cns}
$A \in \mathcal{M}_{n \times n}(\mathbb{R})$ è diagonalizzabile se e solo se:
\begin{enumerate}
    \item $p_A(\lambda)$ ha $n$ radici reali (contate con molteplicità), e
    \item $m_g(\lambda_i) = m_a(\lambda_i)$ per ogni autovalore $\lambda_i$.
\end{enumerate}
\end{theorembox}

\begin{theorembox}[Condizione sufficiente]
Se $A \in \mathcal{M}_{n \times n}(\mathbb{R})$ ha $n$ autovalori reali \emph{distinti}, allora $A$ è diagonalizzabile.
\end{theorembox}

\begin{examplebox}[Diagonalizzazione completa]
Diagonalizziamo $A = \begin{pmatrix} 4 & 1 \\ 2 & 3 \end{pmatrix}$ (dall'esempio precedente).

Abbiamo trovato $\lambda_1 = 2$, $\mathbf{v}_1 = (-1, 2)$ e $\lambda_2 = 5$, $\mathbf{v}_2 = (1, 1)$.
Due autovalori distinti in una matrice $2 \times 2$: $A$ è diagonalizzabile.

La matrice $P$ e la diagonalizzazione:
\[ P = \begin{pmatrix} -1 & 1 \\ 2 & 1 \end{pmatrix}, \quad P^{-1} = \frac{1}{-3}\begin{pmatrix} 1 & -1 \\ -2 & -1 \end{pmatrix} = \begin{pmatrix} -1/3 & 1/3 \\ 2/3 & 1/3 \end{pmatrix} \]
\[ P^{-1}AP = \begin{pmatrix} 2 & 0 \\ 0 & 5 \end{pmatrix} = D. \]
\end{examplebox}

\begin{examplebox}[Matrice non diagonalizzabile]
Consideriamo $A = \begin{pmatrix} 1 & 1 \\ 0 & 1 \end{pmatrix}$. Il polinomio caratteristico è $p_A(\lambda) = (1 - \lambda)^2$, quindi $\lambda = 1$ è l'unico autovalore con $m_a(1) = 2$.

L'autospazio: $(A - I)\mathbf{v} = \mathbf{0}$:
\[ \begin{pmatrix} 0 & 1 \\ 0 & 0 \end{pmatrix} \begin{pmatrix} v_1 \\ v_2 \end{pmatrix} = \begin{pmatrix} 0 \\ 0 \end{pmatrix} \quad \Longrightarrow \quad v_2 = 0. \]
$V_1 = \text{Span}((1,0))$, con $m_g(1) = 1 < 2 = m_a(1)$. La matrice \textbf{non} è diagonalizzabile.
\end{examplebox}

\section[Teorema spettrale]{Matrici simmetriche e teorema spettrale}
\label{sec:teorema-spettrale}

\begin{theorembox}[Teorema spettrale (caso reale)]
\label{thm:spettrale}
Sia $A \in \mathcal{M}_{n \times n}(\mathbb{R})$ una matrice \textbf{simmetrica} ($A = A^T$). Allora:
\begin{enumerate}
    \item Tutti gli autovalori di $A$ sono \textbf{reali}.
    \item Autovettori relativi ad autovalori distinti sono \textbf{ortogonali}.
    \item $A$ è \textbf{ortogonalmente diagonalizzabile}: esiste una matrice ortogonale $Q$ ($Q^T Q = I$) tale che $Q^T A Q = D$.
\end{enumerate}
\end{theorembox}

\begin{examplebox}[Diagonalizzazione ortogonale]
Sia $A = \begin{pmatrix} 2 & 1 \\ 1 & 2 \end{pmatrix}$ (simmetrica). Il polinomio caratteristico: $p_A(\lambda) = (2-\lambda)^2 - 1 = \lambda^2 - 4\lambda + 3 = (\lambda-1)(\lambda-3)$.

Autovalori: $\lambda_1 = 1$, $\lambda_2 = 3$.

Autovettori: $V_1 = \text{Span}((-1,1))$, $V_3 = \text{Span}((1,1))$.

Normalizzando: $\mathbf{q}_1 = \frac{1}{\sqrt{2}}(-1, 1)$, $\mathbf{q}_2 = \frac{1}{\sqrt{2}}(1, 1)$.
\[ Q = \frac{1}{\sqrt{2}}\begin{pmatrix} -1 & 1 \\ 1 & 1 \end{pmatrix}, \quad Q^TAQ = \begin{pmatrix} 1 & 0 \\ 0 & 3 \end{pmatrix}. \]
\end{examplebox}
